\section*{Ключевые знания по теме}

\subsection*{Основные понятия и определения}
\begin{itemize}[label=\textcolor{accent}{\textbullet}, leftmargin=*]
    \item \textbf{Синус угла} в прямоугольном треугольнике: $\sin\alpha = \frac{\text{противолежащий катет}}{\text{гипотенуза}}$
    \item \textbf{Косинус угла}: $\cos\alpha = \frac{\text{прилежащий катет}}{\text{гипотенуза}}$
    \item \textbf{Теорема синусов}: $\frac{a}{\sin A} = \frac{b}{\sin B} = \frac{c}{\sin C} = 2R$, где $R$ — радиус описанной окружности
    \item \textbf{Теорема косинусов}: $c^2 = a^2 + b^2 - 2ab\cos C$
    \item \textbf{Вписанный угол} — угол, вершина которого лежит на окружности, а стороны пересекают окружность
    \item \textbf{Центральный угол} — угол с вершиной в центре окружности
    \item \textbf{Градусная мера дуги} равна градусной мере соответствующего центрального угла
    \item \textbf{Вписанный угол} измеряется половиной дуги, на которую он опирается
\end{itemize}

\subsection*{Свойства окружности}
\begin{itemize}[label=\textcolor{accent}{\textbullet}, leftmargin=*]
    \item Вписанный угол, опирающийся на диаметр, — прямой ($90^\circ$)
    \item Вписанные углы, опирающиеся на одну дугу, равны
    \item Сумма противоположных углов вписанного четырёхугольника равна $180^\circ$
    \item Если в четырёхугольник можно вписать окружность, то суммы противоположных сторон равны
    \item Радиус описанной окружности: $R = \frac{abc}{4S}$, где $S$ — площадь треугольника
\end{itemize}

\subsection*{Методы решения задач}
\begin{enumerate}[label=\arabic*., leftmargin=*]
    \item Для нахождения сторон через углы использовать теорему синусов или косинусов
    \item При наличии окружности применять свойства вписанных и центральных углов
    \item Вводить коэффициент пропорциональности $x$, если известны отношения сторон
    \item Использовать теорему Пифагора в прямоугольных треугольниках
    \item Применять формулу площади: $S = \frac{1}{2}ab\sin C$
    \item Для трапеции с вписанной окружностью: $AB + CD = BC + AD$
\end{enumerate}

\subsection*{Типичные ошибки}
\begin{itemize}[label=\textcolor{accent}{\textbullet}, leftmargin=*]
    \item Путают синус и косинус (противолежащий/прилежащий катет)
    \item Забывают, что вписанный угол равен половине центрального
    \item Не учитывают, что синусу $\frac{1}{2}$ соответствуют два угла: $30^\circ$ и $150^\circ$
    \item Ошибаются при раскрытии скобок в теореме косинусов
    \item Не проверяют, существует ли треугольник с данными сторонами
    \item Забывают про ОДЗ для тригонометрических функций
\end{itemize}

\subsection*{Примеры решения}

\textbf{Пример 1.} В $\triangle ABC$: $\angle C = 90^\circ$, $AB = 5$, $\sin A = 0{,}28$. Найти $BC$.

\textit{Решение:} $\sin A = \frac{BC}{AB} \Rightarrow 0{,}28 = \frac{BC}{5} \Rightarrow BC = 1{,}4$.

\textbf{Пример 2.} В $\triangle ABC$: $\angle C = 90^\circ$, $BC = 3$, $\cos A = \frac{2\sqrt{5}}{5}$. Найти $AC$.

\textit{Решение:} $\cos A = \frac{AC}{AB}$. Пусть $AC = 2\sqrt{5}x$, $AB = 5x$. По теореме Пифагора:
$$(5x)^2 = (2\sqrt{5}x)^2 + 3^2 \Rightarrow 25x^2 = 20x^2 + 9 \Rightarrow 5x^2 = 9 \Rightarrow x = \frac{3}{\sqrt{5}}$$
Тогда $AC = 2\sqrt{5} \cdot \frac{3}{\sqrt{5}} = 6$.

\textbf{Пример 3.} Радиус описанной окружности $R = \sqrt{6}$, хорда $AC = 3$. Найти тупой вписанный угол $\angle ABC$.

\textit{Решение:} По теореме синусов: $\frac{AC}{\sin B} = 2R \Rightarrow \frac{3}{\sin B} = 2\sqrt{6} \Rightarrow \sin B = \frac{3}{2\sqrt{6}} = \frac{\sqrt{3}}{2}$.
Отсюда $B = 60^\circ$ или $B = 120^\circ$. Так как угол тупой, ответ: $120^\circ$.

\section*{Тренировочный блок}

\subsection*{Базовый уровень (3 задания)}
\begin{enumerate}[label=\arabic*., leftmargin=*]
    \item В прямоугольном треугольнике гипотенуза равна 10, а синус одного из острых углов равен 0{,}6. Найдите противолежащий катет.
    
    \item Найдите косинус угла $A$, если в $\triangle ABC$ катеты равны 3 и 4, а $\angle C = 90^\circ$ (угол $A$ лежит против катета 3).
    
    \item Центральный угол равен $100^\circ$. Чему равен вписанный угол, опирающийся на ту же дугу?
\end{enumerate}

\subsection*{Средний уровень (3 задания)}
\begin{enumerate}[label=\arabic*., leftmargin=*, start=4]
    \item В $\triangle ABC$: $\angle C = 90^\circ$, $AB = 13$, $\sin A = \frac{5}{13}$. Найдите $AC$.
    
    \item Стороны треугольника равны 5, 6 и 7. Найдите косинус угла, лежащего против стороны 6.
    
    \item Радиус описанной окружности равен 5, а сторона треугольника равна 6. Найдите синус противолежащего угла.
\end{enumerate}

\subsection*{Уровень выше среднего (3 задания)}
\begin{enumerate}[label=\arabic*., leftmargin=*, start=7]
    \item В трапеции $ABCD$ с основаниями $BC$ и $AD$ точки $E$ и $F$ — середины сторон $BC$ и $AD$. В четырёхугольники $ABEF$ и $ECDF$ можно вписать окружности. Докажите, что трапеция равнобедренная.
    
    \item В $\triangle ABC$ известно: $AB = 7$, $CD = 7$, радиус вписанной в $ABEF$ окружности равен 2{,}5. Найдите радиус описанной около трапеции окружности.
    
    \item Дуга $AB$ имеет градусную меру $102^\circ$, $\angle ACB = 33^\circ$. Найдите $\angle BAE$.
\end{enumerate}

\subsection*{Повышенный уровень (1 задание)}
\begin{enumerate}[label=\arabic*., leftmargin=*, start=10]
    \item В равнобедренной трапеции $ABCD$ ($AB = CD = 7$) сумма оснований равна 14. В четырёхугольник $ABEF$ ($E$, $F$ — середины оснований) вписана окружность радиуса 2{,}5. Найдите радиус окружности, описанной около трапеции $ABCD$.
\end{enumerate}
